\section*{Heaviside Step Function}
\subsection*{Transfer Function}
\[
    \laplace{u(t)} = \frac{1}{s}
\]

\subsection*{Impulse Response}
\[
    \deriv{t}{u(t)} = \delta(t)
\]

\subsection*{Matrix-Vector Form}
\[
    \underline{u}[t] =
    \begin{cases}
        \underline{0} & t < 0 \\
        \underline{1} & t \geq 0
    \end{cases}
\]
