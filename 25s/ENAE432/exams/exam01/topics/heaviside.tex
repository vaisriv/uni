\section*{Heaviside Step Function}
\subsection*{Transfer Function}
\[
    \laplace{\func{u}[t]} = \frac{1}{s}
\]

\subsection*{Impulse Response}
\[
    \deriv{t}{\func{u}[t]} = \func{\delta}[t]
\]

\subsection*{Matrix-Vector Form}
\[
    \func{\underline{u}}[t] =
    \begin{cases}
        \underline{0} & t < 0 \\
        \underline{1} & t \geq 0
    \end{cases}
\]
